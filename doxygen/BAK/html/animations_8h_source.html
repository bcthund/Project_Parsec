321  bool AnimationSys::drawWorld(float fScale, bool bVerticalBB) {
322  for(int iA=0; iA<32; ++iA) {

324 
325  // Play sound first or we may miss it
326  if((running[iA].currentFrame==0) && (animation[running[iA].id].iSample!=999)) audioSys->playSound(animation[running[iA].id].iSample);
327 
328  bool bDraw = true;
329  if(timer.get_ticks()>=(running[iA].ticks+animation[running[iA].id].rate)) {
330  ++running[iA].currentFrame;
331  running[iA].ticks = timer.get_ticks();
332  }
333 
334  if(running[iA].currentFrame >= animation[running[iA].id].frames) {
335  --running[iA].loop;
336  if(running[iA].loop>0) running[iA].currentFrame = 0;
337  else {
338  running[iA].bActive = false;
339  bDraw = false;
340  }
341  }
342 
343  if(bDraw) drawWorld(iA, fScale, bVerticalBB);
344  }
345  }
346  return false;
