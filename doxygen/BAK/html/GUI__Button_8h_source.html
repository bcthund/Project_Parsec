22  * history gets focus back.

230  void Button::updateObjectState(iState eExternState) {
231  // Check for external state change, return early
232  if(*valuePtr && !(this->eObjectState&STATE_ACTIVE))
233  this->eObjectState = STATE_ACTIVE;
234  else if(*valuePtr==*returnState[0].off && (this->eObjectState&STATE_ACTIVE))
235  this->eObjectState = STATE_NONE;
236 
237  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) {
238  eObjectState = eExternState;
239  }
240  else {
241  if(!(eExternState&STATE_UPDATE)) {
242  mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
243  }
244  else this->mState = Core::_Mouse::MOUSE_NONE;
245 
246  if(enabled()) {
247  if(con->buttonType==BUTTON_ONESHOT) {
248  if ( (mState == Core::_Mouse::MOUSE_LEFT) && !bRepeatStatus) {
249  eObjectState = STATE_ACTIVE;
250  bRepeatStatus = true;
251  Sound_PlayOn();
252  }
253  else if (mState == Core::_Mouse::MOUSE_HOVER) {
254  eObjectState = STATE_HOVER;
255  bRepeatStatus = false;
256  }
257  else if (bRepeatStatus) {
258  eObjectState = STATE_NONE;
259  }
260  else if (mState == Core::_Mouse::MOUSE_NONE) {
261  eObjectState = STATE_NONE;
262  bRepeatStatus = false;
263  Sound_PlayOff();
264  }
265  }
266  else if(con->buttonType==BUTTON_TOGGLE) {
267  if(bIsGrouped && (eObjectState&STATE_ACTIVE) && (sGroupObject[con->iGroup]!=name && sGroupObject[con->iGroup]!="")) {
268  eObjectState = STATE_NONE;
269  Sound_AbortState();
270  }
271  else if (mState == Core::_Mouse::MOUSE_LEFT) {
272  if(eObjectState&STATE_ACTIVE) {
273  eObjectState = STATE_NONE;
274  if(sGroupObject[con->iGroup]==name) sGroupObject[con->iGroup] = "";
275  Sound_PlayOff();
276  }
277  else if(eObjectState&STATE_HOVER) {
278  eObjectState = STATE_ACTIVE;
279  sGroupObject[con->iGroup] = name;
280  Sound_PlayOn();
281  }
282  else {
283  // Do not report mouse click if group disabled
284  mState = Core::_Mouse::MOUSE_NONE;
285  }
286  }
287  else if (!(eObjectState&STATE_ACTIVE) && mState == Core::_Mouse::MOUSE_HOVER) {
288  eObjectState = STATE_HOVER;
289  }
290  else if (!(eObjectState&STATE_ACTIVE) && mState == Core::_Mouse::MOUSE_NONE) {
291  eObjectState = STATE_NONE;
292  }
293  }
294  else {
295  if (mState == Core::_Mouse::MOUSE_LEFT_DOWN && !bRepeatStatus) {
296  eObjectState = STATE_ACTIVE;
297  bRepeatStatus = true;
298  Sound_PlayOn();
299 
300  if(con->buttonType==BUTTON_DEBOUNCE) {
301  debounceTimer.start();
302  }
303  }
304  else if (mState == Core::_Mouse::MOUSE_LEFT_DOWN && bRepeatStatus) {
305  eObjectState = STATE_NONE;
306  if(con->buttonType==BUTTON_DEBOUNCE) {
307  if (debounceTimer.split() >= con->debounceTime) eObjectState = STATE_ACTIVE;
308  }
309  }
310  else if (mState == Core::_Mouse::MOUSE_HOVER) {
311  eObjectState = STATE_HOVER;
312  bRepeatStatus = false;
313  }
314  else if (mState == Core::_Mouse::MOUSE_NONE) {
315  eObjectState = STATE_NONE;
316  bRepeatStatus = false;
317  }
318  }
319  }
320  else this->eObjectState = STATE_NONE;
321 
322  // Allow mouse hover at any time (used for tooltips)
323  if(!(eExternState&STATE_UPDATE)) {
324  if(mState==Core::_Mouse::MOUSE_HOVER) eObjectState = eObjectState|STATE_HOVER;
325  else eObjectState = eObjectState&~STATE_HOVER;
326  }
327 
328 // // Allow mouse hover when active TODO: ADD TO ALL OBJECTS
329 // if( (this->eObjectState&STATE_ACTIVE) && (mState==Core::_Mouse::MOUSE_HOVER) ) {
330 // this->eObjectState = this->eObjectState | STATE_HOVER;
331 // }
332 // else if( (this->eObjectState&STATE_ACTIVE) && (this->eObjectState&STATE_HOVER) && (mState!=Core::_Mouse::MOUSE_HOVER) ) {
333 // this->eObjectState = STATE_ACTIVE;
334 // }
335  }
336 
337  if(!enabled()) eObjectState |= STATE_DISABLED;
338  updateValuePtr();
339  }

341  void Button::exec(iState eExternState) {
342  if(bInit) {
343  if(con->visibility && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
344  // Update constraints
345  if(bHasParent) con->exec(*parent);
346  else con->exec();
347 
348  if(!bFocusPresent) {
349  updateObjectState(eExternState);
350  if(con->toolTip.bShow) toolTip.updateObjectState(eObjectState);
351  else toolTip.updateObjectState(STATE_NONE);
352  }
353 
354  // Draw button window
355  win.exec(eObjectState);
356 
357  // Draw button text
358  if(eObjectState&STATE_DISABLED) {
359  if(eObjectState&STATE_HOVER) colors.PushFront(gameVars->pallette.gui.disabled.text.hover);
360  else colors.PushFront(gameVars->pallette.gui.disabled.text.base);
361  }
362  else if(eObjectState&STATE_ACTIVE) colors.PushFront(*con->color.text().active);
363  else if(eObjectState&STATE_HOVER) colors.PushFront(*con->color.text().highlight);
364  else colors.PushFront(*con->color.text().base);
365 
366  textSys->draw(con, name, con->eLabelAnchor);
367  colors.PopFront();
368  }
369  }
