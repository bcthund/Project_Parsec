52  float FractalNoise::eval(Vector2f &p) const
53  {
54  int xi = std::floor(p.x);
55  int yi = std::floor(p.y);
56 
57  float tx = p.x - xi;
58  float ty = p.y - yi;
59 
60  int rx0 = xi & kMaxTableSizeMask;
61  int rx1 = (rx0 + 1) & kMaxTableSizeMask;
62  int ry0 = yi & kMaxTableSizeMask;
63  int ry1 = (ry0 + 1) & kMaxTableSizeMask;
64 
65  // random values at the corners of the cell using permutation table
66  const float & c00 = r[permutationTable[permutationTable[rx0] + ry0]];
67  const float & c10 = r[permutationTable[permutationTable[rx1] + ry0]];
68  const float & c01 = r[permutationTable[permutationTable[rx0] + ry1]];
69  const float & c11 = r[permutationTable[permutationTable[rx1] + ry1]];
70 
71  // remapping of tx and ty using the Smoothstep function
72  float sx = smoothstep(tx);
73  float sy = smoothstep(ty);
74 
75  // linearly interpolate values along the x axis
76  float nx0 = lerp(c00, c10, sx);
77  float nx1 = lerp(c01, c11, sx);
78 
79  // linearly interpolate the nx0/nx1 along they y axis
80  return lerp(nx0, nx1, sy);
