161  void ColorSwatch::init() {
162  if(bHasParent) con->exec(*parent);
163  else con->exec();
164 
165  if(bHasParent) swatch = Window(*parent, name, con);
166  else swatch = Window(name, con);
167  swatch.con->color.back().base = colorPtr;
168  swatch.con->color.back().highlight = colorPtr;
169  swatch.init();
170 
171  if(con->bShowBackground) {
172  con->background.setWidth(100, SIZE_CONSTRAINT_RELATIVE);
173  con->background.setHeight(100, SIZE_CONSTRAINT_RELATIVE);
174 // con->background.setBorder(con->border, con->roundBorder);
175  con->background.setBorder(con->borderNormal, con->borderHover);
176  con->background.setRadius(con->radius);
177  con->background.disablePadding();
178  con->background.enableStipple();
179  con->background.setStipplePattern(&Core::stipple[Core::stipple.STIPPLE_CHECKERBOARD]);
180 
181  background = new Window(*con, name, &con->background);
182  background->init();
183  }
184 
185  if(con->bShowSlider) {
186  float div = con->size.y/4.0f;
187  int sliderHeight = div-3;
188  int sliderSpacing = (div-1)/2.0f;
189 
190  con->slider.control.setHeight(sliderHeight, SIZE_CONSTRAINT_ABSOLUTE);
191 
192  con->slider.setY(sliderSpacing*3);
193  sliderR = new Slider<float>(*con, "R", &colorPtr->r, con->slider);
194  con->slider.setY(sliderSpacing);
195  sliderG = new Slider<float>(*con, "G", &colorPtr->g, con->slider);
196  con->slider.setY(-sliderSpacing);
197  sliderB = new Slider<float>(*con, "B", &colorPtr->b, con->slider);
198  con->slider.setY(-(sliderSpacing*3));
199  sliderA = new Slider<float>(*con, "A", &colorPtr->a, con->slider);
200 
201  sliderR->init();
202  sliderG->init();
203  sliderB->init();
204  sliderA->init();
205  }
206 
207  if(con->bShowLabel) {
208  label = new Label(*con, name, &con->label);
209  label->init();
210  }
211 
212  toolTip.init(*con, name);
213 
214  bInit = true;
215  }
