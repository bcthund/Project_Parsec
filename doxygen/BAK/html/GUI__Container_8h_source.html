790  void GUI_Container::execObjects() {
791  if(bInit) {
792  if(con->visibility) {
793  con->exec();
794 
795  // FIXME: Create a Scissor coordinates stack
796  // [ ] Push onto stack
797  // - Includes enable/disable
798  // - A disable stack does not require x,y,w,h
799  // [ ] Pop from stack
800  // [ ] exec stack
801  // - If nothing on stack, do not enable
802 
803  // Always disable scissor before drawing a container
804  glDisable(GL_SCISSOR_TEST);
805  Object::Window::exec();
806 
807  if(con->bScissor) {
808  int padLR = con->vPadding.left+con->vPadding.right;
809  int padTB = con->vPadding.top+con->vPadding.bottom;
810 
811  int w = con->size.x-padLR,
812  h = con->size.y-padTB;
813 
814  int x = ( (con->pos.x-(con->size.x/2)+con->vPadding.left )) + Core::gameVars->screen.half.x,
815  y = ( (con->pos.y-(con->size.y/2)+con->vPadding.bottom )) + Core::gameVars->screen.half.y;
816 
817  glEnable(GL_SCISSOR_TEST);
818  glScissor(x, y, w, h);
819  }
820 
821  for (auto & window : Window.windows) window->exec();
822  for (auto & label : Label.labels) label->exec();
823  for (auto & button : Button.buttons) button->exec();
824  for (auto & slider : Slider.sliders) slider->exec();
825  for (auto & field : Field.fields) field->exec();
826  for (auto & textArea : TextArea.textAreas) textArea->exec();
827  for (auto & textEdit : TextEdit.textEdits) textEdit->exec();
828  for (auto & colorSwatch : ColorSwatch.colorSwatches) colorSwatch->exec();
829  for (auto & checkBox : CheckBox.checkBoxes) checkBox->exec();
830  for (auto & icon : Icon.icons) icon->exec();
831  for (auto & sprite : Sprite.sprites) sprite->exec();
832  for (auto & progressBar : ProgressBar.progressBars) progressBar->exec();
833  for (auto & comboBox : ComboBox.comboBoxes) comboBox->exec();
834  for (auto & container : containers) container->execObjects();
835  }
836  else {
837  // Update values for non-visible objects, as doing so may update object state conditions
838  for (auto & button : Button.buttons) button->updateNoFocus();
839  for (auto & field : Field.fields) field->updateNoFocus();
840  for (auto & checkBox : CheckBox.checkBoxes) checkBox->updateNoFocus();
841  for (auto & icon : Icon.icons) icon->updateNoFocus();
842  for (auto & textEdit : TextEdit.textEdits) textEdit->updateNoFocus();
843  for (auto & slider : Slider.sliders) slider->updateNoFocus();
844  }
845  }
