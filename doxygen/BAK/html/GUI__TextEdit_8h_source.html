274  void TextEdit::execEdit() {
275  if(bHasFocus) {
276  SDL_PumpEvents();
277  keyboard.event = SDL_GetKeyboardState(NULL);
278  keyboard.update();
279  if (keyboard.keys[SDLK_ESCAPE].bActive) {
280  bHasFocus = false;
281  bFocusPresent = false;
282  eObjectState = STATE_NONE;
283  (*con->text.bufferPtr) = sEditBuffer;
284  }
285 
286  con->text.bOverwrite = keyboard.keys[SDLK_INSERT].bActive;
287 
288  bool bShift = keyboard.keys[SDLK_LSHIFT].bActive || keyboard.keys[SDLK_RSHIFT].bActive;
289  bool bCtrl = keyboard.keys[SDLK_LCTRL].bActive || keyboard.keys[SDLK_RCTRL].bActive;
290  bool bAlt = keyboard.keys[SDLK_LALT].bActive || keyboard.keys[SDLK_RALT].bActive;
291 
292  int iModVal = 1;
293  if(bShift) iModVal = 10;
294 
295  if (keyboard.keys[SDLK_RIGHT].bActive) con->text.iCursorPosition = std::min(con->text.iCursorPosition+iModVal, (int)con->text.bufferPtr->length());
296  if (keyboard.keys[SDLK_LEFT].bActive) con->text.iCursorPosition = std::max(con->text.iCursorPosition-iModVal, 0);
297  if (keyboard.keys[SDLK_HOME].bActive) con->text.iCursorPosition = 0;
298  if (keyboard.keys[SDLK_END].bActive) con->text.iCursorPosition = con->text.bufferPtr->length();
299  if (keyboard.keys[SDLK_UP].bActive) con->text.iCursorPosition = con->text.iUpChars;
300  if (keyboard.keys[SDLK_DOWN].bActive) con->text.iCursorPosition = con->text.iDownChars;
301 
302  keyboard.editBufferAt(con->text.bufferPtr, false, &con->text.iCursorPosition, con->text.bOverwrite);
303 
304  if (keyboard.keys[SDLK_RETURN].bActive || keyboard.keys[SDLK_KP_ENTER].bActive) {
305  if (bAlt) {
306  bHasFocus = false;
307  bFocusPresent = false;
308  eObjectState = STATE_NONE;
309  }
310  else {
311  keyboard.insertBufferAt(con->text.bufferPtr, &con->text.iCursorPosition, '\n', false);
312  }
313  }
314  }

317  void TextEdit::updateObjectState(iState eExternState) {
318  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) {
319  eObjectState = eExternState;
320  }
321  else {
322  // Check button input - Do this first so we can modify colors accordingly.
323 // mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
324  if(!(eExternState&STATE_UPDATE)) {
325  mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
326  }
327  else this->mState = Core::_Mouse::MOUSE_NONE;
328 
329  if(enabled()) {
330  if(!bFocusPresent && mState==Core::_Mouse::MOUSE_LEFT) {
331  bHasFocus = true;
332  bFocusPresent = true;
333  sActiveObject = this->name;
334  eObjectState = STATE_ACTIVE;
335  sEditBuffer = *con->text.bufferPtr;
336  con->text.iCursorPosition = con->text.bufferPtr->length();
337  }
338  else if ( !(eObjectState&STATE_ACTIVE) || ((eObjectState&STATE_ACTIVE) && !bFocusPresent) ) {
339  eObjectState = STATE_NONE;
340  if(sActiveObject == this->name) sActiveObject = "";
341  }
342  }
343  else this->eObjectState = STATE_NONE;
344 
345  // Allow mouse hover at any time (used for tooltips)
346  if(!(eExternState&STATE_UPDATE)) {
347  if(mState==Core::_Mouse::MOUSE_HOVER) eObjectState = eObjectState|STATE_HOVER;
348  else eObjectState = eObjectState&~STATE_HOVER;
349  }
350 
351  }
352 
353  if(!enabled()) eObjectState |= STATE_DISABLED;

356  void TextEdit::exec(iState eExternState) {
357  if(bInit && con->visibility && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
358 
359  // Copy constraints
360  con->text.size.constraint.xAuto = con->size.constraint.xAuto;
361  con->text.size.constraint.yAuto = con->size.constraint.yAuto;
362 
363  if(!bFocusPresent || (sActiveObject==this->name)) updateObjectState(eExternState);
364 
365  // Update constraints
366  if(bHasParent) con->exec(*parent);
367  else con->exec();
368 
369  if(eObjectState&STATE_ACTIVE) con->text.bEnableCursor = true;
370  else con->text.bEnableCursor = false;
371 
372  // Draw button window
373  win.exec(eObjectState);
374 
375  execEdit();
376 
377  // Scroll Buttons
378  if(con->enableScrolling) {
379  // Draw button window
380  if(!bScrollOneShot) {
381  win.con->setWidth(con->size.x-con->iScrollSize, SIZE_CONSTRAINT_ABSOLUTE);
382  bScrollOneShot = true;
383  }
384  win.exec(eObjectState);
385 
386  // Update constraints, needed for autosize and direct external modifications
387  if(con->text.size.isAutoSet()) {
388  text.con->setWidth(con->size.x, SIZE_CONSTRAINT_ABSOLUTE);
389  text.con->setHeight(con->size.y, SIZE_CONSTRAINT_ABSOLUTE);
390  }
391  text.exec(eObjectState);
392 
393  if(con->bShowLabel) label->exec(eObjectState);
394 
395  scrollBarContainer->con->setHeight(con->size.constraint.y, SIZE_CONSTRAINT_ABSOLUTE);
396  scrollBar->con->setMinMax(text.con->iMaxLines, 0);
397  scrollBar->con->setHeight(scrollBarContainer->con->size.y-(2*con->iScrollSize), SIZE_CONSTRAINT_ABSOLUTE);
398 
399  if(!(eObjectState&STATE_DISABLED)) {
400  scrollBarContainer->exec();
401  scrollUp->exec();
402  scrollDown->exec();
403  scrollBar->exec();
404  }
405  else {
406  scrollBarContainer->exec(STATE_DISABLED);
407  scrollUp->exec(STATE_DISABLED);
408  scrollDown->exec(STATE_DISABLED);
409  scrollBar->exec(STATE_DISABLED);
410  }
411 
412  // Check scroll buttons
413  if(scrollUp->getState()) text.con->scrollUp();
414  if(scrollDown->getState()) text.con->scrollDown();
415  }
416  else {
417  // Draw button window
418  if(bScrollOneShot) {
419  win.con->setWidth(con->size.x+20, SIZE_CONSTRAINT_ABSOLUTE);
420  bScrollOneShot = false;
421  }
422  win.exec(eObjectState);
423 
424  // Update constraints, needed for autosize and direct external modifications
425  if(con->text.size.isAutoSet()) {
426  text.con->setWidth(con->size.x, SIZE_CONSTRAINT_ABSOLUTE);
427  text.con->setHeight(con->size.y, SIZE_CONSTRAINT_ABSOLUTE);
428  }
429  text.exec(eObjectState);
430  text.exec(eObjectState);
431 
432  if(con->bShowLabel) label->exec(eObjectState);
433  }
434 
435  // Autosize (Needs some extra padding on textbox, not really sure why but probably due to scrolling implementation)
436  if(con->size.constraint.xAuto) con->setWidth(con->text.size.x, SIZE_CONSTRAINT_ABSOLUTE);
437  if(con->size.constraint.yAuto) con->setHeight(con->text.size.y+gameVars->font.vSize.y, SIZE_CONSTRAINT_ABSOLUTE);
438 
439  }
