142 static float grad(int32_t hash, float x, float y) {
143  const int32_t h = hash & 0x3F; // Convert low 3 bits of hash code
144  const float u = h < 4 ? x : y; // into 8 simple gradient directions,
145  const float v = h < 4 ? y : x;

175 float SimplexNoise::noise(float x) {
176  float n0, n1; // Noise contributions from the two "corners"
177 
178  // No need to skew the input space in 1D
179 
180  // Corners coordinates (nearest integer values):
181  int32_t i0 = fastfloor(x);
182  int32_t i1 = i0 + 1;
183  // Distances to corners (between 0 and 1):
184  float x0 = x - i0;
185  float x1 = x0 - 1.0f;
186 
187  // Calculate the contribution from the first corner
188  float t0 = 1.0f - x0*x0;
189 // if(t0 < 0.0f) t0 = 0.0f; // not possible
190  t0 *= t0;
191  n0 = t0 * t0 * grad(hash(i0), x0);
192 
193  // Calculate the contribution from the second corner
194  float t1 = 1.0f - x1*x1;
195 // if(t1 < 0.0f) t1 = 0.0f; // not possible
196  t1 *= t1;
197  n1 = t1 * t1 * grad(hash(i1), x1);
198 
199  // The maximum value of this noise is 8*(3/4)^4 = 2.53125
200  // A factor of 0.395 scales to fit exactly within [-1,1]
201  return 0.395f * (n0 + n1);
202 }

214 float SimplexNoise::noise(float x, float y) {
215  float n0, n1, n2; // Noise contributions from the three corners
216 
217  // Skewing/Unskewing factors for 2D
218  static const float F2 = 0.366025403f; // F2 = (sqrt(3) - 1) / 2
219  static const float G2 = 0.211324865f; // G2 = (3 - sqrt(3)) / 6 = F2 / (1 + 2 * K)
220 
221  // Skew the input space to determine which simplex cell we're in
222  const float s = (x + y) * F2; // Hairy factor for 2D
223  const float xs = x + s;
224  const float ys = y + s;
225  const int32_t i = fastfloor(xs);
226  const int32_t j = fastfloor(ys);
227 
228  // Unskew the cell origin back to (x,y) space
229  const float t = static_cast<float>(i + j) * G2;
230  const float X0 = i - t;
231  const float Y0 = j - t;
232  const float x0 = x - X0; // The x,y distances from the cell origin
233  const float y0 = y - Y0;
234 
235  // For the 2D case, the simplex shape is an equilateral triangle.
236  // Determine which simplex we are in.
237  int32_t i1, j1; // Offsets for second (middle) corner of simplex in (i,j) coords
238  if (x0 > y0) { // lower triangle, XY order: (0,0)->(1,0)->(1,1)
239  i1 = 1;
240  j1 = 0;
241  } else { // upper triangle, YX order: (0,0)->(0,1)->(1,1)
242  i1 = 0;
243  j1 = 1;
244  }
245 
246  // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
247  // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
248  // c = (3-sqrt(3))/6
249 
250  const float x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y) unskewed coords
251  const float y1 = y0 - j1 + G2;
252  const float x2 = x0 - 1.0f + 2.0f * G2; // Offsets for last corner in (x,y) unskewed coords
253  const float y2 = y0 - 1.0f + 2.0f * G2;
254 
255  // Work out the hashed gradient indices of the three simplex corners
256  const int gi0 = hash(i + hash(j));
257  const int gi1 = hash(i + i1 + hash(j + j1));
258  const int gi2 = hash(i + 1 + hash(j + 1));
259 
260  // Calculate the contribution from the first corner
261  float t0 = 0.5f - x0*x0 - y0*y0;
262  if (t0 < 0.0f) {
263  n0 = 0.0f;
264  } else {
265  t0 *= t0;
266  n0 = t0 * t0 * grad(gi0, x0, y0);
267  }
268 
269  // Calculate the contribution from the second corner
270  float t1 = 0.5f - x1*x1 - y1*y1;
271  if (t1 < 0.0f) {
272  n1 = 0.0f;
273  } else {
274  t1 *= t1;
275  n1 = t1 * t1 * grad(gi1, x1, y1);
276  }
277 
278  // Calculate the contribution from the third corner
279  float t2 = 0.5f - x2*x2 - y2*y2;
280  if (t2 < 0.0f) {
281  n2 = 0.0f;
282  } else {
283  t2 *= t2;
284  n2 = t2 * t2 * grad(gi2, x2, y2);
285  }
286 
287  // Add contributions from each corner to get the final noise value.
288  // The result is scaled to return values in the interval [-1,1].
289  return 45.23065f * (n0 + n1 + n2);
290 }

303  float n0, n1, n2, n3; // Noise contributions from the four corners
304 
305  // Skewing/Unskewing factors for 3D
306  static const float F3 = 1.0f / 3.0f;
307  static const float G3 = 1.0f / 6.0f;
308 
309  // Skew the input space to determine which simplex cell we're in
310  float s = (x + y + z) * F3; // Very nice and simple skew factor for 3D
311  int i = fastfloor(x + s);
312  int j = fastfloor(y + s);
313  int k = fastfloor(z + s);
314  float t = (i + j + k) * G3;
315  float X0 = i - t; // Unskew the cell origin back to (x,y,z) space
316  float Y0 = j - t;
317  float Z0 = k - t;
318  float x0 = x - X0; // The x,y,z distances from the cell origin
319  float y0 = y - Y0;
320  float z0 = z - Z0;
321 
322  // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
323  // Determine which simplex we are in.
324  int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k) coords
325  int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k) coords
326  if (x0 >= y0) {
327  if (y0 >= z0) {
328  i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 1; k2 = 0; // X Y Z order
329  } else if (x0 >= z0) {
330  i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 0; k2 = 1; // X Z Y order

333  }
334  } else { // x0<y0

355  float y3 = y0 - 1.0f + 3.0f * G3;
356  float z3 = z0 - 1.0f + 3.0f * G3;

373  if (t1 < 0) {

378  }

380  if (t2 < 0) {

384  n2 = t2 * t2 * grad(gi2, x2, y2, z2);
