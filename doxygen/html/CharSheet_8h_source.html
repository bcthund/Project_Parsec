203  float GetAngle(bool bRad=false) {
204  float fReturn;
205  float fFix = 0;
206  float fCorrect = -90;
207  //if(bRad) fReturn = atan(last[0]/last[2]);
208  //else fReturn = M_RadToDeg(atan(last[0]/last[2]));
209 
210  if(last[0] < 0.0f && last[2] > 0.0f) fFix = 0;
211  else if(last[0] < 0.0f && last[2] < 0.0f) fFix = 180;
212  else if(last[0] > 0.0f && last[2] < 0.0f) fFix = 180;
213  else if(last[0] > 0.0f && last[2] > 0.0f) fFix = 360;
214 
215  fReturn = fFix - Radians(atan(last[0]/last[2])).toDegrees() + fCorrect;
216  if(bRad) fReturn = Degrees(fReturn).toRadians();
217 
218  if(std::isnan(fReturn)) fReturn = 0.0f;
219 
220  return fReturn;
221  }

226  bool DidMove() {
227  bool bReturn = false;
228  if(last[0] != 0.0f || last[2] != 0.0f) bReturn = true;
229  else bReturn = false;
230  return bReturn;
231  }
