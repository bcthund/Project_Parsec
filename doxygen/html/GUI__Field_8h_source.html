377  template <class T> void Field<T>::updateObjectState(iState eExternState) {
378  checkStatePtr();
379 
380  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) this->eObjectState = eExternState;
381  else {
382 
383  if(!(eExternState&STATE_UPDATE)) {
384  this->mState = Core::mouse->checkInput(gameVars->screen.half.x+this->con->pos.x, gameVars->screen.half.y-this->con->pos.y, this->con->size.x, this->con->size.y);
385  }
386  else this->mState = Core::_Mouse::MOUSE_NONE;
387 
388  if(this->enabled() && this->con->bEditable && !(eExternState&STATE_ACTIVE)) {
389  if(!this->bFocusPresent && (this->mState&Core::_Mouse::MOUSE_LEFT)) {
390  bHasFocus = true;
391  this->bFocusPresent = true;
392  this->sActiveObject = this->name;
393  this->eObjectState = STATE_ACTIVE;
394 
395  execEditStart();
396  }
397  else if ( !(this->eObjectState&STATE_ACTIVE) || ((this->eObjectState&STATE_ACTIVE) && !this->bFocusPresent) ) {
398  this->eObjectState = STATE_NONE;
399  if(this->sActiveObject == this->name) this->sActiveObject = "";
400  }
401  }
402  else this->eObjectState = STATE_NONE;
403  }
404 
405  // Allow mouse hover at any time (used for tooltips)
406  if(!(eExternState&STATE_UPDATE)) {
407  if(this->mState&Core::_Mouse::MOUSE_HOVER) this->eObjectState = this->eObjectState|STATE_HOVER;
408  else this->eObjectState = this->eObjectState&~STATE_HOVER;
409  }
410 
411  if(!this->enabled()) this->eObjectState |= STATE_DISABLED;
412  updateStatePtr();
413  }

418  template <class T> void Field<T>::exec(iState eExternState) {
419  if(this->bInit && this->con->visibility && ((this->parent!=nullptr && this->parent->visibility) || (this->parent==nullptr))) {

426  else this->con->exec();
427 
428  if(!this->bFocusPresent || (this->sActiveObject==this->name)) {
429  updateObjectState(eExternState);
430 
431  if(this->con->toolTip.bShow) this->toolTip.updateObjectState(this->eObjectState);
432  else this->toolTip.updateObjectState(STATE_NONE);
433  }
434 
435  // Draw button window
436  win.exec(this->eObjectState);
437 
438  // Check if value has changed FIXME: This should be in Interactive class in a function somehow
439  if(lastValue != vBuffer) this->bStateChanged = true;
440  lastValue = vBuffer;
441 
442  // Draw button text
443  if((this->eObjectState&STATE_HOVER) && !bHasFocus) colors.PushFront(*this->con->colorText.highlight);
444  else if(this->eObjectState&STATE_ACTIVE) colors.PushFront(*this->con->colorText.active);
445  else colors.PushFront(*this->con->colorText.base);
446 
447  textSys->draw(this->con, vBuffer, CONSTRAIN_CENTER);
448  colors.PopFront();
449  }
450  else {
451  textSys->draw(this->con, vBuffer, CONSTRAIN_CENTER);
452  }
453 
454  // Autosize
455  if(this->con->size.constraint.xAuto) {
456 // int maxSize = (vBuffer.length()*Core::gameVars->font.vSize.x)+(this->con->vPadding.left+this->con->vPadding.right);

458  int maxSize = std::max(this->con->size.constraint.xMin, std::min(this->con->size.constraint.xMax , lengthBuffer ) );
459  this->con->setWidth(maxSize, GUI::SIZE_CONSTRAINT_ABSOLUTE);

462  if(this->con->size.constraint.xAuto || this->con->size.constraint.yAuto) this->con->exec();
463
