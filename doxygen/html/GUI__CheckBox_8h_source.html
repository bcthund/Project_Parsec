252  void CheckBox::updateObjectState(iState eExternState) {
253  // Check for external state change, return early
254  checkStatePtr();
255 
256  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) {
257  eObjectState = eExternState;
258  }
259  else {
260  if(!(eExternState&STATE_UPDATE)) {
261  mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
262  }
263  else this->mState = Core::_Mouse::MOUSE_NONE;
264 
265  if(enabled()) {
266  if(con->bToggle) {
267  if(bIsGrouped && (eObjectState&STATE_ACTIVE) && (Core::groups[con->iGroup].object!=name && Core::groups[con->iGroup].object!="")) {
268  eObjectState = STATE_NONE;
269  Sound_AbortState();
270  }
271  else if (mState&Core::_Mouse::MOUSE_LEFT) {
272  if(eObjectState&STATE_ACTIVE) {
273  if (!bIsGrouped || (bIsGrouped && !Core::groups[con->iGroup].bExclusive)) eObjectState = STATE_NONE;
274  if(bIsGrouped && Core::groups[con->iGroup].object==name) Core::groups[con->iGroup].object = "";
275  Sound_PlayOff();
276  }
277  else if(eObjectState&STATE_HOVER) {
278  eObjectState = STATE_ACTIVE;
279  if(bIsGrouped) Core::groups[con->iGroup].object = name;
280  Sound_PlayOn();
281  }
282  else {
283  // Do not report mouse click if group disabled
284  mState = Core::_Mouse::MOUSE_NONE;
285  }
286  }
287  else if (!(eObjectState&STATE_ACTIVE) && (mState&Core::_Mouse::MOUSE_NONE)) {
288  eObjectState = STATE_NONE;
289  }
290  }
291  }
292  else this->eObjectState = STATE_NONE;
293 
294  // Allow mouse hover at any time (used for tooltips)
295  if(!(eExternState&STATE_UPDATE)) {
296  if(mState&Core::_Mouse::MOUSE_HOVER) eObjectState = eObjectState|STATE_HOVER;
297  else eObjectState = eObjectState&~STATE_HOVER;
298  }
299  }
300 
301  if(!enabled()) eObjectState |= STATE_DISABLED;
302  updateStatePtr();
303  }

305  void CheckBox::exec(iState eExternState) {
306  if(bInit && con->visibility && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
307 
308  updateObjectState(eExternState);
309 
310  if(con->toolTip.bShow) toolTip.updateObjectState(eObjectState);
311  else toolTip.updateObjectState(STATE_NONE);
312 
313  update();
314 
315  // Always set the toggle state
316  if(eObjectState&STATE_ACTIVE) *statePtr = true;
317  else *statePtr = false;
318 
319  box.exec(eObjectState);
320  if(eObjectState&STATE_ACTIVE) check.exec(STATE_ACTIVE);
321 
322  if(con->bShowLabel) label->exec(eObjectState);
323  }
