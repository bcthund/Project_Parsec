296  void ComboBox::updateObjectState(iState eExternState) {
297  // Uncomment if object state can be set externally
298  //checkStatePtr();
299 
300  // Generic state handler
301  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) eObjectState = eExternState;
302  else {
303  if(!(eExternState&STATE_UPDATE)) {
304  Vector2f vPos = con->getScrollPos();
305  Core::_Mouse::iMouseState test1 = Core::mouse->checkInput(gameVars->screen.half.x+vPos.x, gameVars->screen.half.y-vPos.y, con->size.x, con->size.y);
306 
307  vPos = con->itemList.getScrollPos();
308  Core::_Mouse::iMouseState test2 = Core::mouse->checkInput(gameVars->screen.half.x+vPos.x, gameVars->screen.half.y-vPos.y, con->itemList.size.x, con->itemList.size.y);
309 // mState = Core::_Mouse::iMouseState(test1 | test2);
310  mState = test1 | test2;
311  }
312  else mState = Core::_Mouse::MOUSE_NONE;
313 
314  // Object specific logic
315  if(enabled()) {
316  if(mState&Core::_Mouse::MOUSE_HOVER) {
317  updateScrollMouse();
318  }
319  eObjectState = STATE_NONE;
320  }
321  else eObjectState = STATE_NONE;
322  }
323 
324  // Allow mouse hover at any time (used for tooltips)
325  if(!(eExternState&STATE_UPDATE)) {
326  if(mState&Core::_Mouse::MOUSE_HOVER) eObjectState = eObjectState|STATE_HOVER;
327  else eObjectState = eObjectState&~STATE_HOVER;
328  }
329 
330  // Report if mouse is in button space
331  if(!(mState&Core::_Mouse::MOUSE_NONE)) eObjectState = eObjectState|STATE_FOCUS;
332  else eObjectState = eObjectState&~STATE_FOCUS;
333 
334  if(!enabled()) eObjectState |= STATE_DISABLED;
335 
336  // Update the state value according to object state results
337  updateStatePtr();
338  }

381  void ComboBox::exec(iState eExternState) {
382  if(bInit && con->visibility && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
383  if(items.size()>0) {
384  updateScroll();
385 
386  //if(!Base::Interactive_Base::bFocusPresent) updateObjectState(eExternState);
387  updateObjectState(eExternState);
388 
389  if(con->toolTip.bShow) toolTip.updateObjectState(eObjectState);
390  else toolTip.updateObjectState(STATE_NONE);
391 
392  // Update constraints
393  if(bHasParent) con->exec(*parent);
394  else con->exec();
395 
396  // Make sure selected item is active (this is really just visual)
397  items[iSelected]->setState(true);
398 
399  // Update selected item
400 // selectedItem->name = items[iSelected]->name;
401  selectedItem->con->text = items[iSelected]->con->text;
402  selectedItem->exec(STATE_NONE);
403 
404  // Visibility and size
405  itemList->con->visibility = selectedItem->getState();
406 
407  // Manage itemList visibility
408  if(selectedItem->getState() && !bSelecting) bSelecting = true;
409  else if(con->bAutoHide && (!(selectedItem->eObjectState&STATE_FOCUS) && !(itemList->eObjectState&STATE_FOCUS)) && bSelecting) {
410  selectedItem->setState(false);
411  bSelecting = false;
412  }
413  else if(!con->bAutoHide) {
414  if((!(selectedItem->eObjectState&STATE_FOCUS) && !(itemList->eObjectState&STATE_FOCUS)) && bSelecting && Core::mouse->button.held[SDL_BUTTON_LEFT]) {
415  selectedItem->setState(false);
416  bSelecting = false;
417  }
418  }
419 
420  // Draw itemList and items
421  if(bSelecting) {
422 // // Detect scroll button status
423  int iMem = iScrollIndex;
424  if(scrollUp->getState()) { iScrollIndex -= 1; }
425  if(scrollDown->getState()) { iScrollIndex += 1; }
426  if(iMem != iScrollIndex) {
427  // Make sure within limits
428  iScrollIndex = std::max(iScrollIndex, 0);
429  iScrollIndex = std::min(iScrollIndex, items.size()-con->iListLength);
430  }
431 
432  // Draw adjusted itemList window
433  int iHeight = std::min(items.size(), con->iListLength);
434  int iScrollHeight = scrollUp->con->size.y + scrollDown->con->size.y;
435  itemList->con->setHeight(iScrollHeight + ((iHeight)*con->item.size.y+itemList->con->vPadding.top_bottom()), SIZE_CONSTRAINT_ABSOLUTE);
436  itemList->exec(STATE_NONE);
437 
438  // Draw up button
439  int xPos = 0,
440  yPos = 0;
441  scrollUp->setEnabled(iScrollIndex>0);
442  scrollUp->con->setPos(xPos, yPos);
443  scrollUp->exec(STATE_NONE);
444  yPos -= scrollUp->con->size.y;
445 
446  // Draw items
447  for(int n=0; n<items.size(); n++) {
448 
449  int iMaxIndex = iScrollIndex+con->iListLength;
450  if(Core::inRange<int>(n, iScrollIndex, iMaxIndex, false, true)) {
451  items[n]->con->setPos(xPos, yPos);
452  items[n]->exec(STATE_NONE);
453 
454  // Prep for next item
455  yPos -= con->item.size.y;
456  }
457  else items[n]->updateNoFocus();
458 
459  if(items[n]->getState()) iSelected = n;
460  }
461 
462  // Draw down button
463  scrollDown->setEnabled((iScrollIndex+con->iListLength)<items.size());
464  scrollDown->con->setPos(xPos, yPos);
465  scrollDown->exec(STATE_NONE);
466 
467  // Check if value has changed
468  if(iLastValue != iSelected) {
469  bStateChanged = true;
470  }
471  iLastValue = iSelected;
472  }
473 
474  // Autosize
475  if(this->con->size.constraint.xAuto) {
476  int lengthBuffer = 0;
477  for(auto item : items) lengthBuffer = std::max(lengthBuffer, int(item->con->text.length()));
478 
479  // Find out what has larger constraints and use that
480  int maxSizeSelected = int(lengthBuffer*Core::gameVars->font.vSize.x)+(con->vPadding.left+con->vPadding.right);
481  int maxSizeList = int(lengthBuffer*Core::gameVars->font.vSize.x)+(itemList->con->vPadding.left+itemList->con->vPadding.right)+(con->item.vPadding.left+con->item.vPadding.right);
482  int maxSize = std::max(maxSizeSelected, maxSizeList);
483 
484  // Set selected item and list width
485  con->setWidth(maxSize, SIZE_CONSTRAINT_ABSOLUTE);
486  itemList->con->setWidth(maxSize, SIZE_CONSTRAINT_ABSOLUTE);
487  }
488  if(this->con->size.constraint.yAuto) {
489  con->setHeight(Core::gameVars->font.vSize.y+(this->con->vPadding.top+this->con->vPadding.bottom), SIZE_CONSTRAINT_ABSOLUTE);
490  }
491  //if(this->con->size.constraint.xAuto || this->con->size.constraint.yAuto) this->con->exec();
492  }
493  else {
494  selectedItem->con->text = "No Items Available";
495  selectedItem->setState(false);
496  selectedItem->exec(STATE_UPDATE);
497  }
498  }
