160  void _BOUNDING_VOLUME::CalcOBB() {
161  OBB.fWidth = (OBB.vMax.x + fabs(OBB.vMin.x));
162  OBB.fHeight = (OBB.vMax.y + fabs(OBB.vMin.y));
163  OBB.fLength = (OBB.vMax.z + fabs(OBB.vMin.z));
164 
165  OBB.vCenter[0] = ((OBB.vMin.x) + (OBB.fWidth /2.0f));
166  OBB.vCenter[1] = ((OBB.vMin.y) + (OBB.fHeight/2.0f));
167  OBB.vCenter[2] = ((OBB.vMin.z) + (OBB.fLength/2.0f));
168 
169  OBB.vCenter.x *= vScale.x;
170  OBB.vCenter.y *= vScale.y;
171  OBB.vCenter.z *= vScale.z;
172 
173  OBB.fWidth *= vScale.x;
174  OBB.fHeight *= vScale.y;
175  OBB.fLength *= vScale.z;
176 
177  OBB.fWidth += fPadding;
178  OBB.fHeight += fPadding;
179  OBB.fLength += fPadding;
180 
181  OBB.vMinP = (OBB.vMin * vScale) - (fPadding/2);
182  OBB.vMaxP = (OBB.vMax * vScale) + (fPadding/2);
183  }

202  void _BOUNDING_VOLUME::CheckXYZ(Vector3f v) {
203  Vector3f vDist;
204 
205  v.x *=vScale.x;
206  v.y *=vScale.y;
207  v.z *=vScale.z;
208 
209  // Default to using OBB center, as it shouldn't matter
210  vDist = v - OBB.vCenter;
211 
212  float fXZ = (vDist.x*vDist.x)+(vDist.z*vDist.z);
213  float fXYZ = fXZ + (vDist.y*vDist.y);
214 
215  if (fXZ > Cylinder.radius) {
216  Cylinder.radius = fXZ;
217  }
218 
219  if (fXYZ > Sphere.radius) {
220  Sphere.radius = fXYZ;
221  }
222  }

224  void _BOUNDING_VOLUME::Finish(Matrix_System &matrix) {
225  Cylinder.radius = sqrtf(Cylinder.radius) + (fPadding/2);
226  Sphere.radius = sqrtf(Sphere.radius) + (fPadding/2);
227 
228  // Default to using OBB center, as it shouldn't matter
229  vCP = vPosition + OBB.vCenter;
230 
231  float fPQ = OBB.fHeight/2.0f;
232  Cylinder.p = Vector3f(vCP.x, vCP.y + fPQ, vCP.z);
233  Cylinder.q = Vector3f(vCP.x, vCP.y - fPQ, vCP.z);
234 
235  matrix.Push();
236  matrix.SetIdentity();
237  //matrix.Translate( vPosition.x+OBB.vCenter[0], vPosition.y+OBB.vCenter[1], vPosition.z+OBB.vCenter[2] );
238  matrix.Translate( vCP );
239  matrix.Rotate(Degrees(vRotation.x).toRadians(), 1.0, 0.0, 0.0);
240  matrix.Rotate(Degrees(vRotation.y).toRadians(), 0.0, 1.0, 0.0);
241  matrix.Rotate(Degrees(vRotation.z).toRadians(), 0.0, 0.0, 1.0);
242 
243  matrix.Push();
244  matrix.Scale( OBB.fWidth, OBB.fHeight, OBB.fLength);
245  mOBB = matrix.GetModelView();
246  matrix.Pop();
247 
248  matrix.Push();
249  matrix.Scale(Sphere.radius);
250  mSphere = matrix.GetModelView();
251  matrix.Pop();
252 
253  matrix.Push();
254  matrix.Scale(Cylinder.radius, OBB.fHeight-(fPadding/2), Cylinder.radius);
255  mCylinder = matrix.GetModelView();
256  matrix.Pop();
257  matrix.Pop();
258 
259  matrix.Push();
260  matrix.SetIdentity();
261  matrix.Translate( vPosition.x+AABB.vCenter[0], vPosition.y+AABB.vCenter[1], vPosition.z+AABB.vCenter[2] );
262  matrix.Scale( AABB.fWidth, AABB.fHeight, AABB.fLength);
263  mAABB = matrix.GetModelView();
264  matrix.Pop();
265 
266  bFinished = true;
267  }
