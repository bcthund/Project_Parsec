100  Vector3f GetMouseRay(float w, float h, Vector3f rotation) {
101  // Normalized Device Coordinates
102  float dx = (2.0f * x) / w - 1.0f;
103  float dy = 1.0f - (2.0f * y) / h;
104  float dz = 1.0f;
105  Vector3f ray_nds = Vector3f(dx, dy, dz);
106 
107  // Homogeneous Clip Coordinates
108  Vector4f ray_clip = Vector4f(ray_nds.x, ray_nds.y, -1.0, 1.0);
109 
110  // 4D Eye (Camera) Coordinates
111  Matrix44f projection_matrix = matrix->GetProjectionMatrix();
112  Matrix44f inverse_projection = projection_matrix.Inverse();
113  Vector4f ray_eye = inverse_projection * ray_clip;
114  ray_eye = Vector4f(ray_eye.x, ray_eye.y, -1.0, 0.0);
115 
116  // World Coordinates
117  Matrix44f view_matrix = matrix->GetModelView();
118  Matrix44f inverse_view = view_matrix.Inverse();
119  Vector4f ray_wor = (inverse_view * ray_eye);
120 
121  // Apply rotations
122  matrix->Push();
123  matrix->SetIdentity();
124  matrix->Rotate(rotation[0], 1.0, 0.0, 0.0);
125  matrix->Rotate(rotation[1], 0.0, 1.0, 0.0);
126  Matrix44f mRot = matrix->GetRotView();
127  matrix->Pop();
128  Vector4f vTemp = mRot * Vector4f(ray_wor.x, ray_wor.y, ray_wor.z, ray_wor.w);
129 
130  // Normalize and return
131  mouseRay = Vector3f (vTemp.x, vTemp.y, vTemp.z).normal();
132  rays[0].pos = Vector3f(0.0f);
133  rays[0].dir = mouseRay;
134  return mouseRay;
135  }
