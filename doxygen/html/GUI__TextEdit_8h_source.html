296  void TextEdit::execEdit() {
297  if(bHasFocus) {
298  SDL_PumpEvents();
299  keyboard.event = SDL_GetKeyboardState(NULL);
300  keyboard.update();
301  if (keyboard.keys[SDLK_ESCAPE].bActive) {
302  bHasFocus = false;
303  bFocusPresent = false;
304  eObjectState = STATE_NONE;
305  (*con->text.bufferPtr) = sEditBuffer;
306  }
307 
308  con->text.bOverwrite = keyboard.keys[SDLK_INSERT].bActive;
309 
310  bool bShift = keyboard.keys[SDLK_LSHIFT].bActive || keyboard.keys[SDLK_RSHIFT].bActive;
311  bool bCtrl = keyboard.keys[SDLK_LCTRL].bActive || keyboard.keys[SDLK_RCTRL].bActive;
312  bool bAlt = keyboard.keys[SDLK_LALT].bActive || keyboard.keys[SDLK_RALT].bActive;
313 
314  int iModVal = 1;
315  if(bShift) iModVal = 10;
316 
317  if (keyboard.keys[SDLK_RIGHT].bActive) con->text.iCursorPosition = std::min(con->text.iCursorPosition+iModVal, (int)con->text.bufferPtr->length());
318  if (keyboard.keys[SDLK_LEFT].bActive) con->text.iCursorPosition = std::max(con->text.iCursorPosition-iModVal, 0);
319  if (keyboard.keys[SDLK_HOME].bActive) con->text.iCursorPosition = 0;
320  if (keyboard.keys[SDLK_END].bActive) con->text.iCursorPosition = con->text.bufferPtr->length();
321  if (keyboard.keys[SDLK_UP].bActive) con->text.iCursorPosition = con->text.iUpChars;
322  if (keyboard.keys[SDLK_DOWN].bActive) con->text.iCursorPosition = con->text.iDownChars;
323 
324  keyboard.editBufferAt(con->text.bufferPtr, false, &con->text.iCursorPosition, con->text.bOverwrite);
325 
326  if (keyboard.keys[SDLK_RETURN].bActive || keyboard.keys[SDLK_KP_ENTER].bActive) {
327  if (bAlt) {
328  bHasFocus = false;
329  bFocusPresent = false;
330  eObjectState = STATE_NONE;
331  }
332  else {
333  keyboard.insertBufferAt(con->text.bufferPtr, &con->text.iCursorPosition, '\n', false);
334  }
335  }
336  }

339  void TextEdit::updateObjectState(iState eExternState) {
340  checkStatePtr();
341 
342  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) {
343  eObjectState = eExternState;
344  }
345  else {
346  // Check button input - Do this first so we can modify colors accordingly.
347 // mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
348  if(!(eExternState&STATE_UPDATE)) {
349  mState = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
350  }
351  else this->mState = Core::_Mouse::MOUSE_NONE;
352 
353  if(enabled()) {
354  if(!bFocusPresent && (mState&Core::_Mouse::MOUSE_LEFT)) {
355  bHasFocus = true;
356  bFocusPresent = true;
357  sActiveObject = this->name;
358  eObjectState = STATE_ACTIVE;
359  sEditBuffer = *con->text.bufferPtr;
360  con->text.iCursorPosition = con->text.bufferPtr->length();
361  }
362  else if ( !(eObjectState&STATE_ACTIVE) || ((eObjectState&STATE_ACTIVE) && !bFocusPresent) ) {
363  eObjectState = STATE_NONE;
364  if(sActiveObject == this->name) sActiveObject = "";
365  }
366  }
367  else this->eObjectState = STATE_NONE;
368 
369  // Allow mouse hover at any time (used for tooltips)
370  if(!(eExternState&STATE_UPDATE)) {
371  if((mState&Core::_Mouse::MOUSE_HOVER)) eObjectState = eObjectState|STATE_HOVER;
372  else eObjectState = eObjectState&~STATE_HOVER;
373  }
374 
375  }
376 
377  if(!enabled()) eObjectState |= STATE_DISABLED;
378 
379  updateStatePtr();
380  }

382  void TextEdit::exec(iState eExternState) {

384 
385  // Copy constraints
386  con->text.size.constraint.xAuto = con->size.constraint.xAuto;
387  con->text.size.constraint.yAuto = con->size.constraint.yAuto;
388 
389  if(!bFocusPresent || (sActiveObject==this->name)) updateObjectState(eExternState);
390 
391  // Update constraints
392  if(bHasParent) con->exec(*parent);
393  else con->exec();
394 
395  if(eObjectState&STATE_ACTIVE) con->text.bEnableCursor = true;
396  else con->text.bEnableCursor = false;
397 
398  // Draw button window
399  win.exec(eObjectState);
400 
401  execEdit();
402 
403  // Scroll Buttons
404  if(con->enableScrolling) {
405  // Draw button window
406  if(!bScrollOneShot) {
407  win.con->setWidth(con->size.x-con->iScrollSize, SIZE_CONSTRAINT_ABSOLUTE);
408  bScrollOneShot = true;
409  }
410  win.exec(eObjectState);
411 
412  // Update constraints, needed for autosize and direct external modifications
413  if(con->text.size.isAutoSet()) {
414  text.con->setWidth(con->size.x, SIZE_CONSTRAINT_ABSOLUTE);
415  text.con->setHeight(con->size.y, SIZE_CONSTRAINT_ABSOLUTE);
416  }
417  text.exec(eObjectState);
418 
419  if(con->bShowLabel) label->exec(eObjectState);
420 
421  scrollBarContainer->con->setHeight(con->size.constraint.y, SIZE_CONSTRAINT_ABSOLUTE);
422  scrollBar->con->setMinMax(text.con->iMaxLines, 0);
423  scrollBar->con->setHeight(scrollBarContainer->con->size.y-(2*con->iScrollSize), SIZE_CONSTRAINT_ABSOLUTE);
424 
425  if(!(eObjectState&STATE_DISABLED)) {
426  scrollBarContainer->exec();
427  scrollUp->exec();
428  scrollDown->exec();
429  scrollBar->exec();
430  }
431  else {
432  scrollBarContainer->exec(STATE_DISABLED);
433  scrollUp->exec(STATE_DISABLED);
434  scrollDown->exec(STATE_DISABLED);
435  scrollBar->exec(STATE_DISABLED);
436  }
437 
438  // Check scroll buttons

440  if(scrollDown->getState()) text.con->scrollDown();
441  }
442  else {
443  // Draw button window
444  if(bScrollOneShot) {
445  win.con->setWidth(con->size.x+20, SIZE_CONSTRAINT_ABSOLUTE);
446  bScrollOneShot = false;
447  }
448  win.exec(eObjectState);
449 
450  // Update constraints, needed for autosize and direct external modifications
451  if(con->text.size.isAutoSet()) {
452  text.con->setWidth(con->size.x, SIZE_CONSTRAINT_ABSOLUTE);
453  text.con->setHeight(con->size.y, SIZE_CONSTRAINT_ABSOLUTE);
454  }
455  text.exec(eObjectState);
456  text.exec(eObjectState);
457 
458  if(con->bShowLabel) label->exec(eObjectState);
459  }
460 
461  // Autosize (Needs some extra padding on textbox, not really sure why but probably due to scrolling implementation)
462  if(con->size.constraint.xAuto) con->setWidth(con->text.size.x, SIZE_CONSTRAINT_ABSOLUTE);
463  if(con->size.constraint.yAuto) con->setHeight(con->text.size.y+gameVars->font.vSize.y, SIZE_CONSTRAINT_ABSOLUTE);
464 
465  }
