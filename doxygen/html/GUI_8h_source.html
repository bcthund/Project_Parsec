11 /*
12  * Creates GUI instances for particular objects that can be
13  * used as a building block for a full GUI.
14  *

16  * providing "focus" tracking.
17  * - Focus may be achieved with a static INT that
18  * contains an ID. The ID will represent the
19  * active object.
20  * - Focus should be handled with a "history" so when
21  * an object releases focus the previous object in
22  * history gets focus back.

417  void GUI::createGUI(std::string guiName) {
418  Props_Window c;
419  c.setOrigin(CONSTRAIN_CENTER);
420  c.setAnchor(CONSTRAIN_CENTER);
421  c.setPos(0, 0);
422  c.setWidth(Core::gameVars->screen.res.x, SIZE_CONSTRAINT_ABSOLUTE);
423  c.setHeight(Core::gameVars->screen.res.y, SIZE_CONSTRAINT_ABSOLUTE);
424  c.setPadding(0);
425  c.colorBack.base = &colors[colors().Transparent];
426  c.colorBorder.base = &colors[colors().Transparent];
427  c.bNoInput = true;
428 
429  GUI_Container * gui = new GUI_Container(guiName, c);
430  gui->init();
431  guis.push_back(gui);
432  uint id = guis.size()-1;
433  mapGUI.insert(make_pair(guiName, id));
434  setActiveGUI(guiName);
435  }
