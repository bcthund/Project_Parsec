414  float f1, f2, f3, f4, f5;
415  if(this->con->orientation == Core::GUI::SLIDER_HORIZONTAL) {
416  if(this->con->minValue<this->con->maxValue) {
417  f1 = getNormalized(f-Core::gameVars->screen.half.x, minPos, maxPos);
418  }
419  else {
420  f1 = getNormalized(f-Core::gameVars->screen.half.x, maxPos, minPos);
421  }
422  }
423  else if(this->con->orientation == Core::GUI::SLIDER_VERTICAL) {
424  if(this->con->minValue<this->con->maxValue) {
425  f1 = getNormalized(f-Core::gameVars->screen.half.y, minPos, maxPos);
426  }
427  else {
428  f1 = getNormalized(f-Core::gameVars->screen.half.y, maxPos, minPos);
429  }
430  }
431 
432  f2 = spanValue;
433  if(this->con->minValue<this->con->maxValue) {
434  f3 = this->con->minValue;
435  }
436  else {
437  f3 = this->con->maxValue;
438  }
439  f4 = 0.0f;
440  f5 = f1 * f2 + f3 - f4;
441 
442 // return int(f5/this->con->fMinimumStep)*this->con->fMinimumStep; // Constrain value to minimum step size
443  return int(f5/this->con->vStep[0])*this->con->vStep[0]; // Constrain value to minimum step size
444  }

558  template <class T> void Slider<T>::updateObjectState(iState eExternState) {
559  if(eExternState!=STATE_NONE && !(eExternState&STATE_UPDATE)) {
560  this->eObjectState = eExternState;
561  }
562  else {
563  if(!(eExternState&STATE_UPDATE)) {
564  this->mState = Core::mouse->checkInput(gameVars->screen.half.x+this->con->pos.x, gameVars->screen.half.y-this->con->pos.y, std::max(this->con->size.x, this->con->control.size.x), std::max(this->con->size.y, this->con->control.size.y));
565  }
566  else this->mState = Core::_Mouse::MOUSE_NONE;
567 
568  if(this->enabled()) {
569  // Modify slider value by mouse wheel
570  if(this->mState&Core::_Mouse::MOUSE_HOVER) {
571  updateScrollMouse();
572  }
573 
574  if(!this->bHasFocus && ((this->mState&Core::_Mouse::MOUSE_LEFT) || (this->mState&Core::_Mouse::MOUSE_LEFT_DOWN))) {
575  bHasFocus = true;
576  this->bFocusPresent = true;
577  this->sActiveObject = this->name;
578  this->eObjectState = STATE_ACTIVE;
579  Sound_PlayOn();
580  }
581  else if(bHasFocus && (this->eObjectState&STATE_ACTIVE) && Core::mouse->checkState()!=Core::_Mouse::MOUSE_LEFT_DOWN) {
582  bHasFocus = false;
583  this->bFocusPresent = false;
584  if(this->sActiveObject == this->name) this->sActiveObject = "";
585  this->eObjectState = STATE_NONE;
586  Sound_PlayOff();
587  }
588  else if(!bHasFocus && !(this->eObjectState&STATE_ACTIVE) && (this->mState&Core::_Mouse::MOUSE_HOVER)) {
589  this->eObjectState = STATE_HOVER;
590  }
591  else if (!(this->eObjectState&STATE_ACTIVE) || ((this->eObjectState&STATE_ACTIVE) && !this->bFocusPresent) ) {
592  this->eObjectState = STATE_NONE;
593  }
594  }
595  else this->eObjectState = STATE_NONE;
596

598  if(!(eExternState&STATE_UPDATE)) {

609  template <class T> void Slider<T>::exec(iState eExternState) {
610  if(this->bInit && this->con->visibility && ((this->parent!=nullptr && this->parent->visibility) || (this->parent==nullptr))) {
611  /*
612  * Update constraints
613  */
614  if(this->bHasParent) this->con->exec(*this->parent);
615  else this->con->exec();
616 
617  /*
618  * Update slider min/max values
619  * - This only matters if contraints have changed (future proof, resizable windows)
620  */
621  update();
622  if(!this->bFocusPresent || (this->sActiveObject==this->name)) {
623  updateObjectState(eExternState);
624 
625  if(this->con->toolTip.bShow) this->toolTip.updateObjectState(this->eObjectState);
626  else this->toolTip.updateObjectState(STATE_NONE);
627  }
628 
629  /*
630  * Check if slider update in progress according to LAST saved state.
631  * - Requires that min/max values be updated via update()
632  * - If not moving slider then update value in case changed externally
633  */
634  if(this->eObjectState&STATE_ACTIVE) {
635  updateControlPos();
636 // bValueChanged = true;
637  }
638  else setValue();
639 
640  // Check if value has changed
641  if(lastValue != *this->valuePtr) {
642  this->bStateChanged = true;
643  }
644  lastValue = *this->valuePtr;
645 
646  // Draw Slider
647  slider.exec(this->eObjectState);
648 
649  // Draw Control
650  control.con->setPos(currentPos);
651  control.exec(this->eObjectState);
652 
653  // Draw Label
654  if(this->con->bShowLabel) label->exec(this->eObjectState);
655 
656  // Draw Value
657  if(this->con->bShowField) field->exec(this->eObjectState);
658  }
