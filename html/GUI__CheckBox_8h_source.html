157  void CheckBox::exec(eState colorState) {
158  if(bInit && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
159  // Check button input - Do this first so we can modify colors accordingly.
160  state = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
161  if(con->bToggle) {
162  if(bIsGrouped && eObjectState==STATE_ACTIVE && (sActiveButton[con->group]!=name && sActiveButton[con->group]!="")) {
163  eObjectState = STATE_NONE;
164  }
165  else if (state == Core::_Mouse::MOUSE_LEFT) {
166  if(eObjectState==STATE_ACTIVE) {
167  eObjectState = STATE_NONE;
168  if(sActiveButton[con->group]==name) sActiveButton[con->group] = "";
169  }
170  else if(eObjectState==STATE_HOVER) {
171  eObjectState = STATE_ACTIVE;
172  sActiveButton[con->group] = name;
173  }
174  else {
175  // Do not report mouse click if group disabled
176  state = Core::_Mouse::MOUSE_NONE;
177  }
178  }
179  else if (eObjectState!=STATE_ACTIVE && state == Core::_Mouse::MOUSE_HOVER) {
180  eObjectState = STATE_HOVER;
181  }
182  else if (eObjectState!=STATE_ACTIVE && state == Core::_Mouse::MOUSE_NONE) {
183  eObjectState = STATE_NONE;
184  }
185  }
186 
187  box.exec(eObjectState);
188  if(eObjectState==STATE_ACTIVE) check.exec(STATE_ACTIVE);
189  label.exec(eObjectState);
190  }
191  }
192  }
193  }
