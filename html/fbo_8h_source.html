117  void _FBO::CreateFbo(std::string name, uint w, uint h, uint uiMultisample) {
118  if (uiNumBuffers < uiMaxFbos) {
119  siMap.insert(make_pair(name, uiNumBuffers));

122  glEnable(GL_DEPTH_TEST);

125 
126  glGenFramebuffers(1, &fbo[uiNumBuffers].uiFramebuffer);
127  glBindFramebuffer(GL_FRAMEBUFFER, fbo[uiNumBuffers].uiFramebuffer);
128  fbo[id].w = w;

140  void _FBO::AddColorAttachment(std::string name, uint n=0, GLuint filter=GL_LINEAR, GLuint clamp=GL_CLAMP_TO_EDGE) {
141  int id = siMap[name];
142  if(fbo[id].uiMultisample>0) {
143  // std::cout << "Adding Multisampled Color Buffer for " << name << std::endl;
144  glGenRenderbuffers(1, &fbo[id].uiColorBuffer);
145  glBindFramebuffer(GL_FRAMEBUFFER, fbo[id].uiFramebuffer);
146  glBindRenderbuffer(GL_RENDERBUFFER, fbo[id].uiColorBuffer);
147  glRenderbufferStorageMultisample(GL_RENDERBUFFER, fbo[id].uiMultisample, GL_RGBA8, fbo[id].w, fbo[id].h);
148  glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, fbo[id].uiColorBuffer);
149  if (bDebug) std::cout << "\t '" << name << "' Color Buffer: ";
150  fbo[id].bColorBuffer = true;
151  } else {
152  glGenTextures(1, &fbo[id].uiTexColor[n]);
153  glBindTexture(GL_TEXTURE_2D, fbo[id].uiTexColor[n]);
154  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, fbo[id].w, fbo[id].h, 0, GL_RGBA, GL_UNSIGNED_BYTE, 0);
155  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, filter);
156  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, filter);
157  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, clamp);
158  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_R, clamp);
159  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, clamp);
160  uint uiAttach = 0;
161  switch(n) {
162  case 0: uiAttach = GL_COLOR_ATTACHMENT0; break;
163  case 1: uiAttach = GL_COLOR_ATTACHMENT1; break;
164  case 2: uiAttach = GL_COLOR_ATTACHMENT2; break;
165  case 3: uiAttach = GL_COLOR_ATTACHMENT3; break;
166  default: uiAttach = GL_COLOR_ATTACHMENT0;
167  }
168  glFramebufferTexture(GL_FRAMEBUFFER, uiAttach, fbo[id].uiTexColor[n], 0);
169  if (bDebug) std::cout << "\t '" << name << "' Color Texture: ";
170  fbo[id].bTexColor[n] = true;
171  }
172  CheckStatus();
173  }

285  case GL_FRAMEBUFFER_UNSUPPORTED: std::cout << "GL_FRAMEBUFFER_UNSUPPORTED" << std::endl; break;

287  case GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS: std::cout << "GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS" << std::endl; break;
