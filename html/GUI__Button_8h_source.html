251  void Button::exec() {
252  if(bInit && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
253  // Update constraints
254  if(bHasParent) con->exec(*parent);
255  else con->exec();
256 
257  // Check button input - Do this first so we can modify colors accordingly.
258  state = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
259 
260  if(con->bToggle) {
261  if(bIsGrouped && eObjectState==STATE_ACTIVE && (sActiveButton[con->group]!=name && sActiveButton[con->group]!="")) {
262  eObjectState = STATE_NONE;
263  }
264  else if (state == Core::_Mouse::MOUSE_LEFT) {
265  if(eObjectState==STATE_ACTIVE) {
266  eObjectState = STATE_NONE;
267  if(sActiveButton[con->group]==name) sActiveButton[con->group] = "";
268  }
269  else if(eObjectState==STATE_HOVER) {
270  eObjectState = STATE_ACTIVE;
271  sActiveButton[con->group] = name;
272  }
273  else {
274  // Do not report mouse click if group disabled
275  state = Core::_Mouse::MOUSE_NONE;
276  }
277  }
278  else if (eObjectState!=STATE_ACTIVE && state == Core::_Mouse::MOUSE_HOVER) {
279  eObjectState = STATE_HOVER;
280  }
281  else if (eObjectState!=STATE_ACTIVE && state == Core::_Mouse::MOUSE_NONE) {
282  eObjectState = STATE_NONE;
283  }
284  }
285  else {
286  if ( (state == Core::_Mouse::MOUSE_LEFT) && ((con->bNoRepeat && !bRepeatStatus) || !con->bNoRepeat) ) {
287  eObjectState = STATE_ACTIVE;
288  bRepeatStatus = true;
289  }
290  else if (state == Core::_Mouse::MOUSE_HOVER) {
291  eObjectState = STATE_HOVER;
292  bRepeatStatus = false;
293  }
294  else if ( con->bNoRepeat && bRepeatStatus ) {
295  eObjectState = STATE_NONE;
296  }
297  else if (state == Core::_Mouse::MOUSE_NONE) {
298  eObjectState = STATE_NONE;
299  bRepeatStatus = false;
300  }
301  }
302 
303  // Always set the toggle state
304  if(eObjectState==STATE_ACTIVE) *bToggleState = true;
305  else *bToggleState = false;
306 
307  // Draw button window
308  win.exec(eObjectState);
309 
310  // Draw button text
311  if(eObjectState==STATE_ACTIVE) colors->PushFront(*con->colorText.active);
312  else if(eObjectState==STATE_HOVER) colors->PushFront(*con->colorText.highlight);
313  else colors->PushFront(*con->colorText.base);
314 
315 // textSys->draw(*con, name, CONSTRAIN_CENTER);
316  textSys->draw(*con, name, con->eLabelAnchor);
317  colors->PopFront();
318  }
319  }
320  }
321  }
