196  if (bShadow) Core.shader.use(GLS_DIRECTIONAL);

198  for (int count=0; count < o3dData[0].idcount; count++) {
199  Core.matrix.Push();
200  gameVars->texture.sprite.Set(o3dData[0].image[count]);
201  Core.matrix.Translate((o3dData[0].x[count]*32)+16, (o3dData[0].blend[count])+20, (o3dData[0].z[count]*32)+16);
202  Core.matrix.Rotate(M_DegToRad(o3dData[0].rx[count]), 1, 0, 0);
203  Core.matrix.Rotate(M_DegToRad(o3dData[0].ry[count]), 0, 1, 0);
204  Core.matrix.Rotate(M_DegToRad(o3dData[0].rz[count]), 0, 0, 1);
205  Core.matrix.Scale(o3dData[0].sx[count], o3dData[0].sy[count], o3dData[0].sz[count]);
206  Core.matrix.SetTransform();
207  if (bShadow) Core.shader.getUniform(GLS_DIRECTIONAL);
208  else Core.shader.getUniform(GLS_SHADED_TEX);
209  o3dData[0].vao[count].Draw(GLM_DRAW_ELEMENTS);
210  Core.matrix.Pop();
211  }
212  glEnable(GL_CULL_FACE);
