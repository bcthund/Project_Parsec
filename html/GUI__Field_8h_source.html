160  /*
161  * Execute slider logic routine
162  */
163  template <class T> void Field<T>::exec(eState colorState) {
164  if(bInit && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
165 
166  // Update constraints
167  if(bHasParent) con->exec(*parent);
168  else con->exec();
169 
170  // Check button input - Do this first so we can modify colors accordingly.
171  if(colorState==STATE_NONE) {
172  state = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
173  if(!bFocusPresent && state==Core::_Mouse::MOUSE_LEFT) {
174  bHasFocus = true;
175  bFocusPresent = true;
176  eObjectState = STATE_ACTIVE;
177 
178  execEditStart();
179  }
180  else if(!bHasFocus && eObjectState!=STATE_ACTIVE && state==Core::_Mouse::MOUSE_HOVER) eObjectState = STATE_HOVER;
181  else if ( (eObjectState!=STATE_ACTIVE) || ((eObjectState==STATE_ACTIVE) && !bFocusPresent) ) eObjectState = STATE_NONE;
182  }
183  else eObjectState = colorState;
184 
185  // Draw button window
186  win.exec(eObjectState);
187 
188  std::string vBuffer = execEdit();
189 
190  // Draw button text
191  if(eObjectState == STATE_HOVER && !bHasFocus) colors->PushFront(*con->colorText.highlight);
192  else if(eObjectState == STATE_ACTIVE) colors->PushFront(*con->colorText.active);
193  else colors->PushFront(*con->colorText.base);
194 
195  textSys->draw(*con, vBuffer, CONSTRAIN_CENTER);
196  colors->PopFront();
197  }
198  }
