187  void TextEdit::execEdit() {
188  if(bHasFocus) {
189  SDL_PumpEvents();
190  keyboard.event = SDL_GetKeyboardState(NULL);
191  keyboard.update();
192  if (keyboard.keys[SDLK_ESCAPE].bActive) {
193  bHasFocus = false;
194  bFocusPresent = false;
195  eObjectState = STATE_NONE;
196  *con->text.buffer = sEditBuffer; // TESTING: EditBuffer saves previous state
197  }
198 
199  // FIXME: Allow keyboard repeat AFTER a debounce timer, to allow single press and repeats.
200  if (keyboard.keys[SDLK_RIGHT].bActive) con->text.iCursorPosition = std::min(con->text.iCursorPosition+1, (int)con->text.buffer->length());
201  if (keyboard.keys[SDLK_LEFT].bActive) con->text.iCursorPosition = std::max(con->text.iCursorPosition-1, 0);
202  if (keyboard.keys[SDLK_HOME].bActive) con->text.iCursorPosition = 0;
203  if (keyboard.keys[SDLK_END].bActive) con->text.iCursorPosition = con->text.buffer->length();
204 
205  // FIXME: Detect upper and lower line positions? HOW??? Line length varies. Possibly an array of line lengths? Or starting index?
206  if (keyboard.keys[SDLK_UP].bActive) con->text.iCursorPosition = 0;
207  if (keyboard.keys[SDLK_DOWN].bActive) con->text.iCursorPosition = con->text.buffer->length();
208 
209  keyboard.editBuffer(*con->text.buffer, false);
210 
211  if (keyboard.keys[SDLK_RETURN].bActive || keyboard.keys[SDLK_KP_ENTER].bActive) {
212  bHasFocus = false;
213  bFocusPresent = false;
214  eObjectState = STATE_NONE;
215  }
216  }
217  }

219  void TextEdit::exec(eState colorState) {
220  if(bInit && ((parent!=nullptr && parent->visibility) || (parent==nullptr))) {
221  // Update constraints
222  if(bHasParent) con->exec(*parent);
223  else con->exec();
224 
225  // Check button input - Do this first so we can modify colors accordingly.
226  if(colorState==STATE_NONE) {
227  state = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);
228  if(!bFocusPresent && state==Core::_Mouse::MOUSE_LEFT) {
229  bHasFocus = true;
230  bFocusPresent = true;
231  eObjectState = STATE_ACTIVE;
232  sEditBuffer = *con->text.buffer; // TESTING: EditBuffer saves previous state
233  con->text.iCursorPosition = con->text.buffer->length();
234 
235  }
236  else if(!bHasFocus && eObjectState!=STATE_ACTIVE && state==Core::_Mouse::MOUSE_HOVER) eObjectState = STATE_HOVER;
237  else if ( (eObjectState!=STATE_ACTIVE) || ((eObjectState==STATE_ACTIVE) && !bFocusPresent) ) eObjectState = STATE_NONE;
238  }
239  else eObjectState = colorState;
240 
241  if(eObjectState == STATE_ACTIVE) con->text.bEnableCursor = true;
242  else con->text.bEnableCursor = false;
243 
244  // Draw button window
245  win.exec(eObjectState);
246 
247  execEdit();
248 
249  TextArea::exec(eObjectState);
250  }
