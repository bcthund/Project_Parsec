436  template <class T> void Slider<T>::updateScrollMouse() {
437  T modVal = (spanValue/spanPos); // Default, move by 1 pixel worth of value
438 
439  const Uint8 *keyState = SDL_GetKeyboardState(NULL);
440  if (keyState[SDL_SCANCODE_LSHIFT] || keyState[SDL_SCANCODE_RSHIFT]) modVal *= 10;
441  else if (keyState[SDL_SCANCODE_LCTRL] || keyState[SDL_SCANCODE_RCTRL]) modVal /= 10;
442 
443  if(con->minValue>con->maxValue) modVal = -modVal;
444 
445  Core::_Mouse::MOUSE_STATE wheel = Core::mouse->checkWheel();
446  if(wheel == Core::_Mouse::MOUSE_WHEEL_UP) *valuePtr += (modVal*con->scrollSpeed);
447  if(wheel == Core::_Mouse::MOUSE_WHEEL_DOWN) *valuePtr -= (modVal*con->scrollSpeed);
448  }

482  // =-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
483  state = Core::mouse->checkInput(gameVars->screen.half.x+con->pos.x, gameVars->screen.half.y-con->pos.y, con->size.x, con->size.y);

485  /*
486  * Modify slider value by mouse wheel
487  */
488  if(state == Core::_Mouse::MOUSE_HOVER) {
489  updateScrollMouse();
490  }
491

493  bHasFocus = true;
494  bFocusPresent = true;
495  eObjectState = STATE_ACTIVE;
496  bValueChanged = true; // Don't care if value ACTUALLY changed, always assume it did
497  }
498  else if(bHasFocus && eObjectState==STATE_ACTIVE && Core::mouse->checkState()!=Core::_Mouse::MOUSE_LEFT_DOWN) {
499  bHasFocus = false;
500  bFocusPresent = false;
501  eObjectState = STATE_NONE;
502  }
503  else if(!bHasFocus && eObjectState!=STATE_ACTIVE && state==Core::_Mouse::MOUSE_HOVER) eObjectState = STATE_HOVER;
504  else if (eObjectState!=STATE_ACTIVE) eObjectState = STATE_NONE;
505 
506  /*
507  * Check if slider update in progress according to LAST saved state.
508  * - Requires that min/max values be updated via update()
509  * - If not moving slider then update value in case changed externally
510  */
511  if(eObjectState == STATE_ACTIVE) {
512  updateControlPos();
513  bValueChanged = true;
514  }
515  else setValue();
