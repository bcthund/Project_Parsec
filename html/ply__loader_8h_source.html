140  /*
141  * Save the vertex, normal, and coordinate data
142  */

144  std::string * mySplit = split(dataLine[i].c_str(), ' ');
145  vVerts[i][0] = -strtod(mySplit[0].c_str(), nullptr);
146  vVerts[i][1] = strtod(mySplit[2].c_str(), nullptr);
147  vVerts[i][2] = strtod(mySplit[1].c_str(), nullptr);

149  vNorms[i][0] = -strtod(mySplit[3].c_str(), nullptr);
150  vNorms[i][1] = strtod(mySplit[5].c_str(), nullptr);
151  vNorms[i][2] = strtod(mySplit[4].c_str(), nullptr);
152 
153  vCoords[i][0] = strtod(mySplit[6].c_str(), nullptr);
154  vCoords[i][1] = 1-strtod(mySplit[7].c_str(), nullptr); //Correction for blender texcoord difference
155  delete[] mySplit;
156  }
157 
158  /*
159  * Save the index data and convert quads to triangles
160  */
161  count=0;
162  for (uint i=0; i<faceId; i++) {
163  std::string * mySplit = split(indexLine[i].c_str(), ' ');
164 
165  if(stoi(mySplit[0]) == 3) {
166  vIndex[count] = stoi(mySplit[1]);
167  vIndex[count+1] = stoi(mySplit[2]);
168  vIndex[count+2] = stoi(mySplit[3]);
169  }
170  else if(stoi(mySplit[0]) == 4) {
171  face+=1; // Add a face to the loaded list
172 
173  vIndex[count] = stoi(mySplit[1]);
174  vIndex[count+1] = stoi(mySplit[2]);
175  vIndex[count+2] = stoi(mySplit[3]);
176 
177  count+=3;
178 
179  vIndex[count] = stoi(mySplit[1]);
180  vIndex[count+1] = stoi(mySplit[3]);
181  vIndex[count+2] = stoi(mySplit[4]);
182  }
183  count+=3;
184  delete[] mySplit;
185  }
