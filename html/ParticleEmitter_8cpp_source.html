198  void _ParticleEmitter::update(int id) {

200  if ( (timeSys->get_ticks() - emitters[id]->fLastUpdate) > emitters[id]->iRate) {
201

203  emitters[id]->fElapsedEmitterLife = timeSys->get_ticks() - emitters[id]->fStartTime;
204 
205 // if (emitters[id]->sName == "Sparks") {

208 // }

211 
212  bool bChange = false;
213 
214  // ==============================
215  // Remove and Move particles
216  // ------------------------------
217  // Update particles from top down so index doesn't become invalid when a dead particle is erased
218  for (int n=emitters[id]->particles.size()-1; n>=0; n--) {
219

222

231  // Particle rising
232  emitters[id]->remove(n);
233  bChange = true;
234  }
235  else if ( (emitters[id]->fGravityEffect > 0) && (emitters[id]->particles[n].vPosition.y < emitters[id]->fClip) ) {
236  // Particle Falling

263  // -------------------------

267  bChange = true;
268  }
269 
270  emitters[id]->fLastEmit = timeSys->get_ticks();
271  }
272 
273  // ========================
274  // Sort particles by distance
275  // Calculate alpha
276  // Calculate size
277  // ------------------------
278  for (int n=0; n<emitters[id]->particles.size(); n++) {
279  Vector3f vDirection = emitters[id]->particles[n].vPosition + gameVars->player.active->transform.pos;
280  emitters[id]->particles[n].fDistance = vDirection.length(); // MUST be actual distance so we can calcualte scale based on distance later
281 
282  float fAlpha = 1.0f;
283  if (emitters[id]->fParticleLifeTime <= 0.0f) {
284  if (emitters[id]->fGravityEffect <= 0) {
285  // Particle rising
286  fAlpha = 1.0f - ( emitters[id]->particles[n].vPosition.y/(emitters[id]->fClip-100.0f) );
287  }
288  else {
289  // Particle Falling
290  fAlpha = emitters[id]->particles[n].vPosition.y/(emitters[id]->fClip+100.0f);
291  }
292  }
293  else {
294  fAlpha = 1.0f - (emitters[id]->particles[n].fElapsedLife / emitters[id]->particles[n].fLifeTime);
295  }
296  emitters[id]->particles[n].vColor.a = fAlpha * fLifeFactor;
297 
298  // Scale particle with distance
299  float a = 0.0f;
300  float b = 0.0f;
301  float c = 1.0f/gameVars->screen.fFar;
302  float d = emitters[id]->particles[n].fDistance;
303  float d2 = d*d;
304  float fSize = sqrt(1/(a+b*d+c*d2));
305  emitters[id]->particles[n].fSize = fmax(fSize * emitters[id]->fStartingSize, 1.0f);
306  }
307  std::sort(emitters[id]->particles.begin(), emitters[id]->particles.end(), compareParticleDistance());
308 
309  emitters[id]->fLastUpdate = timeSys->get_ticks();
310  }
311  else {
312  remove(id);
313  }

337  void _ParticleEmitter::emit(int id) {
338  _ParticleData newParticle;
339  newParticle.fElapsedLife = 0;
340  newParticle.fStartTime = timeSys->get_ticks();
341  newParticle.fGravityEffect = emitters[id]->fGravityEffect + (emitters[id]->fGravityEffect * roll(emitters[id]->vGravityVariation));
342  newParticle.fLifeTime = (emitters[id]->fParticleLifeTime + (emitters[id]->fParticleLifeTime * roll(emitters[id]->vLifeVariation)));
343  newParticle.fRotation = emitters[id]->fRotation + (emitters[id]->fRotation * roll(emitters[id]->vRotationVariation));
344  newParticle.fRotationSpeed = emitters[id]->fRotationSpeed + (emitters[id]->fRotationSpeed * roll(emitters[id]->vRotationSpeedVariation));
345  newParticle.fSize = emitters[id]->fStartingSize + (emitters[id]->fStartingSize * roll(emitters[id]->vSizeVariation));
346  newParticle.vColor = Color(1.0f);
347  newParticle.vPosition = *emitters[id]->vSource + Vector3f( roll(emitters[id]->vSpawnSpread.x),
348  roll(emitters[id]->vSpawnSpread.y),
349  roll(emitters[id]->vSpawnSpread.z) );
350 
351  Vector3f velocity = (*emitters[id]->vTarget-*emitters[id]->vSource);
352  velocity = velocity.normal();
353  velocity += Vector3f( roll(emitters[id]->vVelocitySpread.x),
354  roll(emitters[id]->vVelocitySpread.y),
355  roll(emitters[id]->vVelocitySpread.z) );
356  velocity *= emitters[id]->fSpeed;
357  newParticle.vVelocity = velocity;
358 
359  emitters[id]->add(newParticle);
360  move(id, emitters[id]->particles.size()-1); // Make first move now, this will update distance for sorting
361  }
