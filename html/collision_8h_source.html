132  bool _Collision::Ray_AABB ( Vector3f p, Vector3f d, _BOUNDING_VOLUME &a, _BOUNDING_VOLUME::_BB &b, bool bUpdateCData) {
133  const float EPSILON = 0.0001;
134  float tmin = 0.0f;
135  float tmax = 1024.0f;
136  for(int i=0; i<3; i++) {
137  if(fabs(d[i]) < EPSILON) {
138  if(p[i] < b.vMinP[i] || p[i] > b.vMaxP[i]) goto ReturnFalse;
139  }
140  else {
141  float ood = 1.0f / d[i];
142  float t1 = (b.vMinP[i] - p[i]) * ood;
143  float t2 = (b.vMaxP[i] - p[i]) * ood;
144 
145  if(t1>t2) std::swap(t1,t2);
146 
147  tmin = std::max(tmin, t1);
148  tmax = std::min(tmax, t2);
149 
150  if(tmin>tmax) goto ReturnFalse;
151  }
152  }
153 
154  if(bUpdateCData) SetCData(a, CalcIntersection(p, d, tmin), tmin, true);
155  else a.cdata.t = tmin; // Always save this in case OBB needs it
156  return true;
157 
158  ReturnFalse:
159  if(bUpdateCData) SetCData(a, Vector3f(0.0f), 0.0f, false);
160  return false;

163  bool _Collision::Ray_Sphere(Vector3f p, Vector3f d, _BOUNDING_VOLUME &s) {
164 // std::cout << "Doing Collistion Test:" << std::endl;
165 // std::cout << "\t p = " << p.x << ", " << p.y << ", " << p.z << std::endl;
166 // std::cout << "\t d = " << d.x << ", " << d.y << ", " << d.z << std::endl;

170 
171  float discr = 0.0f,
172  t = 0.0f;
173 
174  Vector3f m = (p-(*s.vTranslate)) - s.GetCenterPosition();
175  float b = Core::gmath.Dot(m, d);
176  float c = Core::gmath.Dot(m, m) - (s.Sphere.radius * s.Sphere.radius);
177 
178  if(c > 0.0f && b > 0.0f) goto ReturnFalse;
179  discr = b * b - c;
180 
181  if(discr < 0.0f) goto ReturnFalse;
182 
183  t = -b - std::sqrt(discr);
184 
185  if(t < 0.0f) t = 0.0f;
186 
187  // Ray has hit sphere otherwise
188  SetCData(s, CalcIntersection(p, d, t), t, true);
189  return true;
190 
191  ReturnFalse:
192  SetCData(s, Vector3f(0.0f), 0.0f, false);
193  return false;

196  bool _Collision::Ray_SphereFast(Vector3f p, Vector3f d, _BOUNDING_VOLUME &s) {
197 // std::cout << "Doing Collistion Test:" << std::endl;
198 // std::cout << "\t p = " << p.x << ", " << p.y << ", " << p.z << std::endl;
199 // std::cout << "\t d = " << d.x << ", " << d.y << ", " << d.z << std::endl;

201 // std::cout << "\ts.cp = " << s.vCP.x << ", " << s.vCP.y << ", " << s.vCP.z << std::endl;
202 // std::cout << "\t s.r = " << s.fSphereRadius << std::endl << std::endl;
203 
204  Vector3f m = (p-(*s.vTranslate)) - s.GetCenterPosition();
205  float c = Core::gmath.Dot(m, m) - (s.Sphere.radius * s.Sphere.radius);
206 
207  // At least one real root, intersection exists
208  if(c <= 0.0f) return true;
209 
210  float b = Core::gmath.Dot(m, d);
211 
212  // Ray origin outside sphere, pointing away
213  if(b > 0.0f) return false;
214 
215  float disc = b*b - c;
216 
217  // Negative discriminant means ray misses sphere
218  if(disc < 0.0f) return false;
219 
220  // Ray has hit sphere otherwise
221  return true;

224  bool _Collision::Ray_Cylinder(Vector3f vP, Vector3f vD, _BOUNDING_VOLUME &bv, bool bUpdateCData) {
225  float fRayLength = 1000,
226  r = bv.Cylinder.radius,
227  EPSILON = 0.0001f,
228  t = 0.0f;
229 
230  Vector3f sa = vP - (*bv.vTranslate),
231  sb = sa + (vD*fRayLength),
232  p = bv.Cylinder.p,
233  q = bv.Cylinder.q,
234  d = q - p,
235  m = sa - p,
236  n = sb - sa;
237 
238  bool bReturn = false;
239 
240  // Cross-initialization prevention
241  float nn,
242  mn,
243  a,
244  k,
245  c,
246  b,
247  discr,
248  t0;
249 
250 // std::cout << "Testing Cylinder:" << std::endl;
251 // std::cout << "\t sa:" << sa.x << ", " << sb.y << ", " << sa.z << std::endl;
252 // std::cout << "\t sb:" << sb.x << ", " << sa.y << ", " << sb.z << std::endl;
253 // std::cout << "\t p:" << p.x << ", " << p.y << ", " << p.z << std::endl;
254 // std::cout << "\t q:" << q.x << ", " << q.y << ", " << q.z << std::endl;
255 // std::cout << "\t d:" << d.x << ", " << d.y << ", " << d.z << std::endl;
256 // std::cout << "\t m:" << m.x << ", " << m.y << ", " << m.z << std::endl;
257 // std::cout << "\t n:" << n.x << ", " << n.y << ", " << n.z << std::endl;
258 // std::cout << "\t r:" << r << std::endl << std::endl;
259 
260  float md = Core::gmath.Dot(m, d);
261  float nd = Core::gmath.Dot(n, d);
262  float dd = Core::gmath.Dot(d, d);
263 
264  // Segment fully outside of endcap
265  if(md < 0.0f && md + nd < 0.0f) goto ReturnFalse; // Outside p ide
266  if(md > dd && md + nd > dd) goto ReturnFalse; // Outside q side
267 
268  nn = Core::gmath.Dot(n, n);
269  mn = Core::gmath.Dot(m, n);
270  a = dd * nn - nd * nd;
271  k = Core::gmath.Dot(m, m) - (r * r);
272  c = dd * k - md * md;
273 
274  if(std::fabs(a) < EPSILON) {
275  // Segment runs parallel to axis
276  if(c > 0.0f) goto ReturnFalse;
277  // Segment intersects cylinder
278  if(md < 0.0f) t = -mn / nn; // Intersect segment against p endcap
279  else if(md > dd) t = (nd - mn) / nn; // Intersect segment against q endcap
280  else t = 0.0f; // 'a' inside cylinder
281  if(bUpdateCData) SetCData(bv, CalcIntersection(vP, vD*fRayLength, t), t, true);
282  return true;
283  }
284 
285  b = dd * mn - nd * md;
286  discr = b * b - a * c;
287  if(discr < 0.0f) goto ReturnFalse;
288 
289  t0 = t = (-b - std::sqrt(discr)) / a;
290  // Intersection lies outside segment (commented for RAY intersection)
291  //if(t < 0.0f || t > 1.0f) { std::cout << " -* 4" << std::endl; return false; }
292  if(md + t * nd < 0.0f) { // Intersection outside p-side cylinder
293  if(nd <= 0.0f) goto ReturnFalse;
294  t = -md / nd;
295  bReturn = k + t * (2.0f * mn + t * nn) <= 0.0f;
296  if(bUpdateCData) SetCData(bv, CalcIntersection(vP, vD*fRayLength, t), t, bReturn);
297  return bReturn;
298  }
299  else if(md + t * nd > dd) { // Intersection outside q-side cylinder
300  if(nd >= 0.0f) goto ReturnFalse;
301  t = (dd - md) / nd;
302  bReturn = k + dd - 2.0f * md + t * (2.0f * (mn - nd) + t * nn) <= 0.0f;
303  if(bUpdateCData) SetCData(bv, CalcIntersection(vP, vD*fRayLength, t), t, bReturn);
304  return bReturn;
305  }
306  t = t0;
307  bReturn = t >= 0.0f && t <= 1.0f;
308  if(bUpdateCData) SetCData(bv, CalcIntersection(vP, vD*fRayLength, t), t, bReturn);
309  return bReturn;
310 
311  ReturnFalse:
312  if(bUpdateCData) SetCData(bv, Vector3f(0.0f), 0.0f, false);
313  return false;
314  }
